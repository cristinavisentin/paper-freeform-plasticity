$lastIndexesXor = [-40:1:-1]        % last 40 (=10%)
$firstIndexesXor = [0:1:39]         % first 40 (=10%)

$qFunctions = f.allNamed(fs = m.sKeyMap(
  keys = [maturePerformance; increase; inconsistency];
  values = [
    ds.e.sBool.avgScore(
      scoreType = boolean;
      caseIndexes = $lastIndexesXor
    );
    ea.f.numExpr(
      expr = "-(x2;x1)";
      args = [
        ds.e.sBool.avgScore(
          scoreType = unlimited;
          caseIndexes = $firstIndexesXor
        );
        ds.e.sBool.avgScore(
          scoreType = unlimited;
          caseIndexes = $lastIndexesXor
        )
      ]
    );
    ds.e.sBool.sdScore(
      scoreType = unlimited;
      indexes = $lastIndexesXor
    )
  ]
))

$xorProblem = ea.p.simsToTo(
  name = "xor";
  simulations = 5 * [ds.s.sequentialXor(
    cases = $casesXor;
    rewardType = unlimited;
    resetAgent = true;
    shuffle = true
  )];
  dT = 1;
  tRange = $tRangeXor;
  comparator = ea.comparator.descending(of = f.mapValue(key = maturePerformance));
  qFunction = $qFunctions;
  aggregator = f.mapAggregate(aggregateF = f.avg())
)
