$lastIndexesMopm3 = [-192:1:-1]     % last 192 (=10%)
$firstIndexesMopm3 = [0:1:191]      % last 192 (=10%)

$qFunctions = f.allNamed(fs = m.sKeyMap(
  keys = [maturePerformance; increase; inconsistency];
  values = [
    ds.e.sBool.avgScore(
      scoreType = boolean;
      caseIndexes = $lastIndexesMopm3
    );
    ea.f.numExpr(
      expr = "-(x2;x1)";
      args = [
        ds.e.sBool.avgScore(
          scoreType = unlimited;
          caseIndexes = $firstIndexesMopm3
        );
        ds.e.sBool.avgScore(
          scoreType = unlimited;
          caseIndexes = $lastIndexesMopm3
        )
      ]
    );
    ds.e.sBool.sdScore(
      scoreType = unlimited;
      indexes = $lastIndexesMopm3
    )
  ]
))

$mopmProblem = ea.p.simsToTo(
  name = "mopm";
  simulations = 5 * [ds.s.sequentialBf(
    cases = $casesMopm3;
    target = ea.misc.mopmBf(n = 3);
    rewardType = unlimited;
    resetAgent = true
  )];
  dT = 1;
  tRange = $tRangeMopm3;
  comparator = ea.comparator.descending(of = f.mapValue(key = maturePerformance));
  qFunction = $qFunctions;
  aggregator = f.mapAggregate(aggregateF = f.avg())
)
